# 一、需要包含的头文件

#include <glad/glad.h>
#include <GLFW/glfw3.h>           //第一行与第二行顺序不能交换
#include <iostream>


const unsigned int SCR_WIDTH = 800;  //  设置窗口大小的全局常量（见三）
const unsigned int SCR_HEIGHT = 600;

# 二、创建 main 函数实例化 GLFW 窗口

int main()
{
    glfwInit();   //调用glfwInit函数初始化glfw

    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);  //设置主要和次要版本
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);

    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE); //设置为使用核心模式

    //glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);  //Mac OS X系统需要这行代码进行转换，Windows无需此行；

    return 0;
}

# 三、创建一个窗口对象（写在 main 函数中）

GLFWwindow* window = glfwCreateWindow(SCR_WIDTH, SCR_HEIGHT, "LearnOpenGL", NULL, NULL);  //这里的  SCR_WIDTH, SCR_HEIGHT 为窗口的大小，需要定义全局常量（见一）；
  if (window == NULL)
  {
    std::cout << "Failed to create GLFW window" << std::endl;  
    glfwTerminate();    //结束线程
    return -1;
  }

# 四、初始化 GLAD（写在 main 函数中）

GLAD是用来管理OpenGL的函数指针的，所以在调用任何OpenGL的函数之前我们需要初始化GLAD。

if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress))
{
  std::cout << "Failed to initialize GLAD" << std::endl;
  return -1;
}

# 五、视口

在我们开始渲染之前还有一件重要的事情要做，我们必须告诉OpenGL渲染窗口的尺寸大小，即视口(Viewport)，这样OpenGL才只能知道怎样根据窗口大小显示数据和坐标。

# 5.1 定义一个回调函数（写在 mian 函数之外）

）
void framebuffer_size_callback (GLFWwindow* window, int width, int height) // 每当窗口改变大小（手动拉改变窗口大小），GLFW 会调用这个函数并填充相应的参数供你处理。
{
glViewport (0, 0, width, height); // 确保视口与新窗口尺寸匹配；请注意，宽度和高度将明显大于视网膜显示器上指定的宽度和高度。
}

# 5.2 注册这个函数（写在 main 函数中）

告诉GLFW我们希望每当窗口调整大小的时候调用这个函数：
  
glfwSetFramebufferSizeCallback(window, framebuffer_size_callback);   //注册窗口监听函数：注册这个函数，告诉GLFW我们希望每当窗口调整大小的时候调用framebuffer_size_callback这个函数；

# 六、渲染循环 & 输入 & 渲染

我们可不希望只绘制一个图像之后我们的应用程序就立即退出并关闭窗口。我们希望程序在我们主动关闭它之前不断绘制图像并能够接受用户输入。

# 6.1 渲染循环 & 输入 & 渲染

while(!glfwWindowShouldClose(window))       //渲染循环（写在main函数之中）
{
  processInput(window);  //输入；在渲染循环的每一个迭代中调用processInput函数，判断是否按下ESC键；

  glClearColor(0.2f, 0.3f, 0.3f, 1.0f);  //渲染，调用glClearColor函数来设置清空屏幕后所填充的颜色；
  glClear(GL_COLOR_BUFFER_BIT);  //渲染，调用glClear函数来清空屏幕的颜色缓冲，否则仍能看见上一次迭代的渲染结果；

  glfwSwapBuffers(window);
  glfwPollEvents();    
}

# 6.2 输入函数 processInput

void processInput(GLFWwindow *window)      //processInput函数（写在main函数之外）
{
  if(glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
      glfwSetWindowShouldClose(window, true);
}

# 七、清理退出（写在 main 函数最后

glfwTerminate();  //当渲染循环结束后我们使用glfwTerminate函数正确释放/删除之前的分配的所有资源。
return 0;
